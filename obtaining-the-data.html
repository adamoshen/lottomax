1.2.1 Try 2021 

url2021 <- "https://www.lottomaxnumbers.com/numbers/2021"

nums2021 <- read_html(url2021) %>%
  html_element(css="table") %>%
  html_table(header=TRUE)

1.2.2 General case 

getnums <- function(year){
  yearurl <- paste0("https://www.lottomaxnumbers.com/numbers/", year)
  
  read_html(yearurl) %>%
    html_element(css="table") %>%
    html_table(header=TRUE)
}
